42  public virtual IEnumerable<UserProfile> Filter(UserProfileIndexModel model, ClaimsPrincipal userPrincipal, bool userIsAdmin)
43  {
44  IEnumerable<UserProfile> users;
45  if (userIsAdmin)
46  {
47  if (model.DepartmentId != null)
48  {
49  var department = _departmentRepository.Get(d => d.Id == model.DepartmentId.Value);
50  users = department != null
51  ? GetPage(department.Staff, model.CurrentPage, model.PageSize)
52  : GetPage(model.CurrentPage, model.PageSize);
53  }
54  else
55  {
56  users = GetPage(model.CurrentPage, model.PageSize);
57  }
58  }
59  else
60  {
61  var currentUser = Get(userPrincipal);
62  var department = _departmentRepository.Get(u => u.Head.Id == currentUser.Id);
63  users = GetPage(department.Staff, model.CurrentPage, model.PageSize);
64  }
65 
66  if (model.IsApproved != null)
67  {
68  switch (model.IsApproved.Value)
69  {
70  case UserProfileIndexModel.IsApprovedOption.All:
71  break;
72  case UserProfileIndexModel.IsApprovedOption.Yes:
73  users = users.Where(u => u.IsApproved);
74  break;
75  case UserProfileIndexModel.IsApprovedOption.No:
76  users = users.Where(u => !u.IsApproved);
77  break;
78  }
79  }
80 
81  if (model.FirstName != null)
82  {
83  users = users.Where(u => u.FirstName.ToLower().Contains(model.FirstName.Trim().ToLower()));
84  }
85 
86  if (model.LastName != null)
87  {
88  users = users.Where(u => u.LastName.ToLower().Contains(model.LastName.Trim().ToLower()));
89  }
90 
91  return users;
92  }

180  public virtual async Task<string> ChangePassword(UserProfile user, string oldPassword, string newPassword, string newPasswordRepeat, UserManager<UserProfile> userManager)
181  {
182  string result = null;
183 
184  // Checks if entered password is equal to current user password
185  if (await userManager.CheckPasswordAsync(user, oldPassword))
186  {
187  // Checks if new password and repeated password are equal
188  if (newPassword.Equals(newPasswordRepeat))
189  {
190  // Checks if new password is equal to current user password
191  if (!oldPassword.Equals(newPassword))
192  {
193  var passwordValidator = new PasswordValidator<UserProfile>();
194 
195  // Validates new password
196  var validationResult = await passwordValidator.ValidateAsync(userManager, user, newPassword);
197  if (validationResult.Succeeded)
198  {
199  await userManager.ChangePasswordAsync(user, oldPassword, newPassword);
200  }
201  else
202  {
203  result = validationResult.ToString();
204  }
205  }
206  else
207  {
208  result = "New password must differ from an old one";
209  }
210  }
211  else
212  {
213  result = "Repeated password does not mach a new one";
214  }
215  }
216  else
217  {
218  result = "Old password is incorrect";
219  }
